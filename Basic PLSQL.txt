6.
--PLSQL program for creating subtypes
DECLARE
SUBTYPE NAME IS CHAR(20);
SUBTYPE GREETINGS IS VARCHAR2(30);
SALUTATION NAME;
MESSAGE GREETINGS;
BEGIN
SALUTATION:='SIDDHARTH';
MESSAGE:='HELLO HOW ARE YOU?';
DBMS_OUTPUT.PUT_LINE(SALUTATION);
DBMS_OUTPUT.PUT_LINE(MESSAGE);
END;

--PLSQL program for scope of variables

--GLOBAL
N1 NUMBER;
N2 NUMBER;
BEGIN
N1:=1;
N2:=2;
DBMS_OUTPUT.PUT_LINE('GLOBAL1'|| ' '|| N1);

   BEGIN
