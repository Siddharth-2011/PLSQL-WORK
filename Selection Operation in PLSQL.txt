1.
--select operation in PLSQL
DECLARE
SAL NUMBER;
BEGIN
SELECT SALARY INTO SAL FROM EMPLOYE3 WHERE EMPID=29;
DBMS_OUTPUT.PUT_LINE(SAL);
END;

2.
--selection and updation operation in PLSQL
DECLARE
SAL NUMBER;
NEWSAL NUMBER;
BEGIN
SELECT SALARY INTO SAL FROM EMPLOYE3 WHERE EMPID=39;
NEWSAL:=2*SAL;
UPDATE EMPLOYE3 SET SALARY= NEWSAL WHERE EMPID=39 ;
DBMS_OUTPUT.PUT_LINE(SAL);
END;

3.
--selection operation of many attribute values
DECLARE
SAL NUMBER;
NID NUMBER;
NNAME VARCHAR2(10);
BEGIN
SELECT SALARY, EMPID, ENAME INTO SAL, NID, NNAME FROM EMPLOYE3 WHERE LOCATION='JODHPUR';
DBMS_OUTPUT.PUT_LINE(SAL|| '  '|| NID|| ' ' ||NNAME);
END;

4.
--selection operation using record
DECLARE
RECORD EMPLOYE3%ROWTYPE;
BEGIN
SELECT * INTO RECORD FROM EMPLOYE3 WHERE LOCATION='JODHPUR';
DBMS_OUTPUT.PUT_LINE(RECORD.SALARY);
END;

5.
--selection operation using record
DECLARE
RECORD EMPLOYE3%ROWTYPE;
NSAL NUMBER;
BEGIN
SELECT * INTO RECORD FROM EMPLOYE3 WHERE LOCATION='JODHPUR';
NSAL:=RECORD.SALARY+1000;
UPDATE EMPLOYE3 SET SALARY= NSAL WHERE LOCATION='JODHPUR';
DBMS_OUTPUT.PUT_LINE(NSAL);
END;
