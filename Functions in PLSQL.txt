--function for addition of two numbers
DECLARE
N1 NUMBER;
N2 NUMBER;
N3 NUMBER;
FUNCTION ADD(N1 NUMBER,N2 NUMBER)
RETURN NUMBER IS NUM3 NUMBER;
BEGIN
N3:=N1+N2;
RETURN(N3);
END ADD;
BEGIN
N1:=1;
N2:=2;
N3:=ADD(N1,N2);
DBMS_OUTPUT.PUT_LINE(N3);
END;

2.
--function for addition of two numbers
CREATE OR REPLACE FUNCTION ADD2
(NUM1 IN NUMBER, NUM2 IN NUMBER)
RETURN NUMBER IS NUM3 NUMBER;
BEGIN
NUM3:=NUM1+NUM2;
RETURN(NUM3);
END ADD2; 

DECLARE
RESULT NUMBER;
BEGIN
RESULT:=ADD2(4,4);
DBMS_OUTPUT.PUT_LINE(RESULT);
END;
