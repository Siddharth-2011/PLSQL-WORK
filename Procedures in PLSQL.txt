1.
--procedure to do multiplication of two numbers
DECLARE
N1 NUMBER;
N2 NUMBER;
N3 NUMBER; 
       PROCEDURE MULTI1(N1 IN NUMBER, N2 IN NUMBER, N3 OUT NUMBER) IS 
           BEGIN
           N3:=N1*N2;
       END MULTI1;
BEGIN
N1:=2;
N2:=3;
MULTI1(2,2,N3);
DBMS_OUTPUT.PUT_LINE(N3);
MULTI1(N1,N2,N3);
DBMS_OUTPUT.PUT_LINE(N3);
N1:= :ENTER_NUM1;
N2:= :ENTER_NUM2;
MULTI1(N1,N2,N3);
DBMS_OUTPUT.PUT_LINE(N3);
END;

3.
--procedure to increase salary of employes
DECLARE
AMOUNT NUMBER;
EID NUMBER;
     PROCEDURE RAISESAL(ENO NUMBER, INCREASE NUMBER) IS
     
     BEGIN
     UPDATE EMPLOYE3 SET SALARY=SALARY + INCREASE WHERE EMPID=ENO;
     END RAISESAL;
BEGIN
RAISESAL(39,4000);
END;

4.
--procedure to add two numbers
CREATE OR REPLACE PROCEDURE ADD1
(N1 NUMBER, N2 NUMBER, N3 OUT NUMBER) IS
BEGIN
N3:=N1+N2;
END ADD1;

DECLARE
N3 NUMBER;
BEGIN
ADD1(2,10,N3);
DBMS_OUTPUT.PUT_LINE(N3);
END;
